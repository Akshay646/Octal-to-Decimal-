//convert octal to decimal.

#include <bits/stdc++.h>

using namespace std;

int conv(int n){
    int i = 0, con = 0, temp;
    while(n > 0){
        temp = n % 10;
        con = con + temp * pow(8, i);
        n/=10;
        i++;
    }
    return con;
}

int main()
{
	int n;
	cin >> n;
	
	cout << conv(n);

	return 0;
}
